pipeline {
    agent {
  label 'jenkins-master'
         }
options {
  timestamps()
  buildDiscarder logRotator(artifactDaysToKeepStr: '', artifactNumToKeepStr: '', daysToKeepStr: '2', numToKeepStr: '3')
}

    stages {
        stage('Compile') {
            steps {
		powershell 'maven compile -DVer=$Version'
            }
        }
        
        
        stage('Junit') {
            steps {
		bat 'maven test -DVer=$Version'
                  }
        }
        stage('Package') {
            steps {
		bat 'maven -DskipTests clean package -DVer=$Version'
                  }
        }

        stage('Upload package into Nexus') {
            steps {
		bat 'maven -DskipTests deploy -DVer=$Version'
                  }
        }        
        stage('Cleanup Workspace') {
            steps {
                cleanWs()
            }
        }
    }
}
